Selenium Web Scraper

Use the following Python code to scrape text from a webpage using Selenium.

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys

file = open('movies name.txt', 'w+')

driver = webdriver.Chrome()

for i in range(2000, 2024):
    url = "https://yekmdl1.online/Anime/Movies/" + str(i)
    driver.get(url)
    file.write(driver.find_element(By.XPATH, "/html/body/pre").text + '\n')
