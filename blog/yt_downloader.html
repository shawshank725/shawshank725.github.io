YouTube Video Downloader

Use the following Python code to download YouTube videos by providing the video URL.

url = input('Enter URL - ')

import yt_dlp as youtube_dl
from yt_dlp.postprocessor import FFmpegPostProcessor
FFmpegPostProcessor._ffmpeg_location.set(R'path-to-ffmpeg-module-in-python-folder')

ydl_opts = {}
with youtube_dl.YoutubeDL(ydl_opts) as ydl:
    ydl.download([url])
